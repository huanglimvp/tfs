124  int64_t get_last_read_pri_ds() const
125  {
126  int32_t index = 0;
127  if (PRI_DS_NOT_INIT != pri_ds_index_) // read
128  {
129  // pri_ds_index_ < 0, server is the last retry one
130  index = PRI_DS_TRY_ALL_OVER == pri_ds_index_ ? get_orig_pri_ds_index() : pri_ds_index_;
131  index = index == 0 ? ds_.size() - 1 : index - 1;
132  }
133  return ds_[index];
134  }
