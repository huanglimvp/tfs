120  template<typename Strategy>
121  class StoreWeight
122  {
123  public:
124  StoreWeight(Strategy& strategy, DS_WEIGHT& weight) :
125  strategy_(strategy), weights_(weight) {}
126  virtual ~StoreWeight() {}
127  void operator()(const ServerCollect* server) const
128  {
129  int64_t weight = strategy_.calc(server);
130  TBSYS_LOG(DEBUG, "server: %s weight: %"PRI64_PREFIX"d", tbsys::CNetUtil::addrToString(server->id()).c_str(), weight);
131  if (weight > 0)
132  weights_.insert(std::make_pair(weight, const_cast<ServerCollect*> (server)));
133  }
134  const DS_WEIGHT& get_weight() const
135  {
136  return weights_;
137  }
138  private:
139  Strategy& strategy_;
140  DS_WEIGHT& weights_;
141  };

163  int32_t elect_ds(Strategy& strategy, ElectType op, LayoutManager& meta, std::vector<ServerCollect*>& source,
164  std::vector<ServerCollect*>& except, int32_t elect_count, bool check_server_in_plan, std::vector<ServerCollect*>& result)
165  {
166  DS_WEIGHT weights;
167  StoreWeight<Strategy> store(strategy, weights);
168  std::vector<ServerCollect*>::iterator iter = source.begin();
169  for (; iter != source.end(); ++iter)
170  {
171  //if (!exist(except, (*iter)))
172  bool has_valid = check_server_in_plan ? ((!meta.find_server_in_plan((*iter)))
173  && (!exist(except, (*iter)))) : (!exist(except, (*iter)));
174  if (has_valid)
175  {
176  common::RWLock::Lock lock((*(*iter)), common::READ_LOCKER);
177  store(*iter);
178  }
179  }
180 
181  return op(weights, elect_count, result);

184  template<typename Strategy, typename ElectType>
185  int32_t elect_ds(Strategy& strategy, ElectType op, LayoutManager& meta, std::vector<ServerCollect*>& except,
186  int32_t elect_count, bool check_server_in_plan, std::vector<ServerCollect*> & result)
187  {
188  bool has_valid = false;
189  DS_WEIGHT weights;
190  StoreWeight<Strategy> store(strategy, weights);
191  const SERVER_MAP& ds_map = meta.servers_;
192  SERVER_MAP::const_iterator iter = ds_map.begin();
193  for (; iter != ds_map.end(); ++iter)
194  {
195  has_valid = check_server_in_plan ? ((!meta.find_server_in_plan(iter->second))
196  && (!exist(except, iter->second))) : (!exist(except, iter->second));
197  if(has_valid)
198  {
199  common::RWLock::Lock lock((*iter->second), common::READ_LOCKER);
200  store(iter->second);
201  }
202  }
203  TBSYS_LOG(INFO, "plan hold server size: %u, server size: %u", meta.server_to_task_.size(), meta.servers_.size());
204 
205  return op(weights, elect_count, result);
206  }
