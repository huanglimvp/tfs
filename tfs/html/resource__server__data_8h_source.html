151  void add(const common::SessionStat& s_stat)
152  {
153  const std::map<common::OperType, common::AppOperInfo>& app_oper_info = s_stat.app_oper_info_;
154  std::map<common::OperType, common::AppOperInfo>::const_iterator sit = app_oper_info.begin();
155  for ( ; sit != app_oper_info.end(); ++sit)
156  {
157  if (common::OPER_WRITE == sit->first || common::OPER_UNIQUE_WRITE == sit->first)
158  {
159  file_count_ += sit->second.oper_times_;
160  used_capacity_ += sit->second.oper_size_;
161  }
162  else if (common::OPER_UNLINK == sit->first || common::OPER_UNIQUE_UNLINK == sit->first)
163  {
164  file_count_ -= sit->second.oper_times_;
165  used_capacity_ -= sit->second.oper_size_;
166  }
167  }
168  }
169  };
